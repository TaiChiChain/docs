</ParamField>

<ParamField body="jsonrpc" type="string" placeholder="2.0">

</ParamField>

</ParamField>

<ParamField body="params" type="array">
    Contains the list of parameters passed to the remote method
</ParamField>

### Response

<ResponseField name="id" type="number">
    Returns unique identifier for the request
</ResponseField>

<ResponseField name="jsonrpc" type="string">
    Returns the version of JSON-RPC being used
</ResponseField>


<ResponseField name="inspectResponse" type="object">
    A list of pending and queued transactions(format with textual data) in txpool

        <Expandable title="inspectResponse object">
            <ResponseField name="pending" type="array">
                An array of transaction objects, with textual data
                <Expandable title="pending array">
                    <ResponseField name="address" type="array">
                        The address initiating a transaction
                        <Expandable title="pool transactions response object">
                            <ResponseField name="nonce" type="array">
                                The nonce of the sending address
                                <Expandable title="textual transaction object">
                                    <ResponseField name="textualData" type="string">
                                        The textual data of the transaction
                                    </ResponseField>
                                </Expandable>
                            </ResponseField>
                        </Expandable>
                    </ResponseField>
                </Expandable>
            </ResponseField>
            <ResponseField name="queued" type="array">
                An array of transaction objects, with textual data
                <Expandable title="queued array">
                    <ResponseField name="address" type="array">
                        The address initiating a transaction
                        <Expandable title="pool transactions response object">
                            <ResponseField name="nonce" type="array">
                                The nonce of the sending address
                                <Expandable title="textual transaction object">
                                    <ResponseField name="textualData" type="string">
                                        The textual data of the transaction
                                    </ResponseField>
                                </Expandable>
                            </ResponseField>
                        </Expandable>
                    </ResponseField>
                </Expandable>
            </ResponseField>
        </Expandable>
</ResponseField>
