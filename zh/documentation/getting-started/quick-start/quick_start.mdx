这份指南将向您介绍如何快速接入和使用 AxiomLedger 区块链，您可以十分方便地在 AxiomLedger 区块链上发起交易、铸造代币，也可以通过 AxiomLedger 的区块链浏览器查看区块和交易的信息，相信这将对您后续在 AxiomLedger 上进行开发有一定的帮助作用。
**以下步骤都将以 Gemini 测试网为例，如果您想在主网或其他测试网上尝试，可在[资源](/zh/documentation/getting-started/resources/resources)页查看其他链的相关资源。**

## 使用钱包连接 AxiomLedger

AxiomLedger 是兼容 EVM 的区块链，任何您熟悉的以太坊工具都可以在 AxiomLedger 上使用，例如 `Truffle`、`Remix` 和 `ether.js` 等。您只需要拥有一个基于以太坊的账户地址，就可以通过主流的钱包（例如 Metamask 等）或 SDK 连接到 AxiomLedger 的测试网。

如果您是第一次使用 MetaMask，请点击[此处](https://metamask.io/download/)安装插件。如果您是第一次连接 AxiomLedger，您需要将 RPC 地址、链 ID、代币符号等信息新增到网络列表。主网及测试网的链信息可在[资源](/zh/documentation/getting-started/resources/resources)页查看。

连接到 AxiomLedger 后，一般来说您的账户余额是 `0` AXC，此时您可以访问 AxiomLedger 测试网水龙头，在页面上填入您的账户地址来获取一些 AXC。

**_请注意：在 24 小时内，一个账户只能在水龙头获取一次 AXC！_**

## 构造和发起您的第一笔交易

如果您需要深入了解 AxiomLedger，想获取更多链上信息或发起链上交易，推荐您使用 Axiom 提供的官方 SDK （axiom.js），它的用法与[ether.js](https://github.com/ethers-io/ethers.js)类似。

复制以下代码，将其保存为后缀为.mjs 的文件，根据您的测试账户替换转账方的私钥以及接收方的地址（即代码中的 account_from 和 account_to），然后执行 `node YourFileName.mjs`（请确保已提前安装 Node.js）。等待终端返回交易哈希即代表交易已成功，此时可查看 Metamask 中的余额是否更新。若网络有延迟，可稍后查询。可以在[AXMScan](https://scan.gemini.axiomesh.io/)中粘贴返回的交易哈希查看交易详情。

<Warning>
  需注意：若您的电脑上启动了代理服务可能导致交易失败，可关闭代理后再次尝试。
</Warning>

```Solidity
// Javascript
// Setup: npm install @axiomesh/axiom
import { ethers } from '@axiomesh/axiom'

const provider = new ethers.JsonRpcProvider("https://rpc1.gemini.axiomesh.io");

async function main() {
  // query the BlockNumber on AxiomLedger
  const latestBlock = await provider.getBlockNumber();
  console.log("The latest block number is", latestBlock);

  // send transaction to AxiomLedger : transfer AXC
  // step1. Create account variables
  const account_from = {
    privateKey: 'YOUR-PRIVATE-KEY-HERE',
  };
  const account_to = {
    address: 'ADDRESS-TO-HERE',
  };

  // step2. Create wallet
  const wallet = new ethers.Wallet(account_from.privateKey, provider);
  // step3. Create tx object
  const tx = {
    to: account_to.address,
    value: ethers.parseEther('0.1'),
  };
  // step4. Sign and send tx - wait for receipt
  const createReceipt = await wallet.sendTransaction(tx);
  await createReceipt.wait();
  console.log("Transaction successful with hash:", createReceipt.hash);
}

main();
```

## 铸造一个 Token

相信您在之前的步骤中成功连接到了 AxiomLedger，也体验了 AxiomLedger 官方 SDK 的基本用法，接下来可以探索在 AxiomLedger 上铸造您自己的 Token。一般来说，这需要您提前编写或准备好一个符合 ERC20 标准的智能合约，并将它成功部署到 AxiomLedger 上，然后在钱包软件（例如 Metamask）上添加合约地址，就可以查看到您新铸造的 Token 了。
为了降低您的使用难度，AxiomLedger 提供了一份简单的智能合约样本，您只需要修改其中的 Token 名称（样本以 TAXM 为例）和账户地址即可。当然，您也可以参考此合约开发更符合您需要的智能合约。

```solidity
pragma solidity ^0.4.24;

//Safe Math Interface

contract SafeMath {

    function safeAdd(uint a, uint b) public pure returns (uint c) {
        c = a + b;
        require(c >= a);
    }

    function safeSub(uint a, uint b) public pure returns (uint c) {
        require(b <= a);
        c = a - b;
    }

    function safeMul(uint a, uint b) public pure returns (uint c) {
        c = a * b;
        require(a == 0 || c / a == b);
    }

    function safeDiv(uint a, uint b) public pure returns (uint c) {
        require(b > 0);
        c = a / b;
    }
}


//ERC Token Standard #20 Interface

contract ERC20Interface {
    function totalSupply() public constant returns (uint);
    function balanceOf(address tokenOwner) public constant returns (uint balance);
    function allowance(address tokenOwner, address spender) public constant returns (uint remaining);
    function transfer(address to, uint tokens) public returns (bool success);
    function approve(address spender, uint tokens) public returns (bool success);
    function transferFrom(address from, address to, uint tokens) public returns (bool success);

    event Transfer(address indexed from, address indexed to, uint tokens);
    event Approval(address indexed tokenOwner, address indexed spender, uint tokens);
}


//Contract function to receive approval and execute function in one call

contract ApproveAndCallFallBack {
    function receiveApproval(address from, uint256 tokens, address token, bytes data) public;
}

//Actual token contract

contract TAXMToken is ERC20Interface, SafeMath {
    string public symbol;
    string public  name;
    uint8 public decimals;
    uint public _totalSupply;

    mapping(address => uint) balances;
    mapping(address => mapping(address => uint)) allowed;

    constructor() public {
        symbol = "TAXM";  // Replace the symbol with your own
        name = "TAXM Coin";  // Replace the name with your own
        decimals = 2;
        _totalSupply = 10000000;
        balances[0xc7F999b83Af6DF9e67d0a37Ee7e900bF38b3D013] = _totalSupply;  // Replace the address with your own
        emit Transfer(address(0), 0xc7F999b83Af6DF9e67d0a37Ee7e900bF38b3D013, _totalSupply); // Replace the address with your own
    }

    function totalSupply() public constant returns (uint) {
        return _totalSupply  - balances[address(0)];
    }

    function balanceOf(address tokenOwner) public constant returns (uint balance) {
        return balances[tokenOwner];
    }

    function transfer(address to, uint tokens) public returns (bool success) {
        balances[msg.sender] = safeSub(balances[msg.sender], tokens);
        balances[to] = safeAdd(balances[to], tokens);
        emit Transfer(msg.sender, to, tokens);
        return true;
    }

    function approve(address spender, uint tokens) public returns (bool success) {
        allowed[msg.sender][spender] = tokens;
        emit Approval(msg.sender, spender, tokens);
        return true;
    }

    function transferFrom(address from, address to, uint tokens) public returns (bool success) {
        balances[from] = safeSub(balances[from], tokens);
        allowed[from][msg.sender] = safeSub(allowed[from][msg.sender], tokens);
        balances[to] = safeAdd(balances[to], tokens);
        emit Transfer(from, to, tokens);
        return true;
    }

    function allowance(address tokenOwner, address spender) public constant returns (uint remaining) {
        return allowed[tokenOwner][spender];
    }

    function approveAndCall(address spender, uint tokens, bytes data) public returns (bool success) {
        allowed[msg.sender][spender] = tokens;
        emit Approval(msg.sender, spender, tokens);
        ApproveAndCallFallBack(spender).receiveApproval(msg.sender, tokens, this, data);
        return true;
    }

    function () public payable {
        revert();
    }
}
```

如果您已经对以太坊有一定的了解，可以通过 Remix 编译并部署此智能合约。请注意：

- 编译合约时请选择合适的 Solidity 版本(^0.4.24)
- 通过 injected Provider - MetaMask 的方式部署，请确认网络已经切换到 AxiomLedger 测试网并且账户地址符合您的选择
- 已选择能正确生成 Token 的合约(例如该例中的`TestToken`)

## 完成一笔 Token 转账

在上一步您已经完成了铸造 Token，现在可以试试 Token 转账的操作。同样，为了更直观地完成和展示操作，推荐您选择 Remix+MetaMask 的方式去转账。首先创建一个新的账户，其地址用来接收 Token，然后在刚才部署合约的页面下方，调用`transfer`方法（需要传入新账户的地址和转账的金额），在 MetaMask 弹框中确认发出交易。

## 体验 AxiomLedger 区块链浏览器
