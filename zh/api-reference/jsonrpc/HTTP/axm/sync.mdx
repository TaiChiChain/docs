---
title: "axm_syncProgress"
api: "POST /"
---


返回节点的同步进度。


### Body

<ParamField body="id" type="number" placeholder="1">
  请求标识符
</ParamField>

<ParamField body="jsonrpc" type="string" placeholder="2.0">
  正在使用的JSON-RPC版本
</ParamField>

<ParamField body="method" type="string" placeholder="debug_traceTransaction">
  进行调用的远程方法名称
</ParamField>

<ParamField body="params" type="array">
   包含传递给远程方法的参数列表
</ParamField>


### Response

<ResponseField name="id" type="number">
  返回请求的唯一标识符
</ResponseField>

<ResponseField name="jsonrpc" type="string">
  返回正在使用的JSON-RPC版本
</ResponseField>



<ResponseField name="syncProgress" type="object">
    一个同步对象

    <Expandable title="syncProgress object">
        <ResponseField name="inSync" type="bool">
            一个布尔值，当节点不在同步中或同步结束时为 false（同步所有区块但未在账本中追赶）。
        </ResponseField>

        <ResponseField name="catchUp" type="bool">
            一个布尔值，当节点不在同步中或在账本中追赶结束时为 true。
        </ResponseField>

        <ResponseField name="startSyncBlock" type="string">
           当前同步期开始时的区块编号的十六进制代码
        </ResponseField>

        <ResponseField name="currentSyncHeight" type="string">
            当前同步期中当前同步块高度的区块编号的十六进制代码
        </ResponseField>

        <ResponseField name="highestBlockHeight" type="string">
            当前账本最高区块高度的区块编号的十六进制代码
        </ResponseField>

        <ResponseField name="currentSyncHeight" type="string">
            当前同步期中目标高度的区块编号的十六进制代码
        </ResponseField>

        <ResponseField name="syncMode" type="string">
            指示特定同步模式的代码，完整模式或快照模式
        </ResponseField>

        <ResponseField name="peers" type="array">
            此同步期间的远程节点列表
        </ResponseField>

    </Expandable>



</ResponseField>
