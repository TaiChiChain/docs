---
title: "logs"
---

日志订阅类型发出与指定主题筛选器匹配的日志，这些日志包含在新添加的块中

### Body

<ParamField body="id" type="number" placeholder="1">
  日志订阅类型发出与指定主题筛选器匹配的日志，这些日志包含在新添加的块中
</ParamField>

<ParamField body="jsonrpc" type="string" placeholder="2.0">
  正在使用的JSON-RPC版本
</ParamField>

<ParamField body="method" type="string" placeholder="eth_subscribe">
  进行调用的远程方法名称(订阅时统一为eth_subscribe)
</ParamField>

<ParamField body="params" type="array">
  包含传递给远程方法的参数列表

  <Expandable title="params">
     表示用于启动订阅的方法和参数
     <ParamField body="subscribe methods" type="string" placeholder="logs">
        logs-subscribe返回包含在新导入的块中并符合给定筛选条件的日志
     </ParamField>

    <Expandable title="logs params">
        <ParamField body="logs subscribe address" type="string">
            单一地址或地址数组。只有从其中一个地址创建的日志才会发出
        </ParamField>

        <ParamField body="logs subscribe topics" type="array">
            主题说明符数组
        </ParamField>
    </Expandable>


  </Expandable>
  
</ParamField>



### Response

<ResponseField name="jsonrpc" type="string">
  返回正在使用的JSON-RPC版本
</ResponseField>

<ResponseField name="method" type="string">
  返回JSON-RPC调用的方法
</ResponseField>

<ResponseField name="params" type="object">
  订阅结果

  <Expandable title="subscription result">

    <ResponseField name="subscription" type="string">
      订阅Id
    </ResponseField>

    <Expandable title="result">
        <ResponseField name="address" type="string">
            address属性指定此日志的来源地址。
        </ResponseField>

        <ResponseField name="blockHash" type="string">
            The blockHash property specifies the hash of the block in which the transaction was included.
        </ResponseField>

        <ResponseField name="blockNumber" type="string">
            blockNumber属性指定包含事务的块的编号。它被编码为十六进制字符串。
        </ResponseField>

        <ResponseField name="data" type="string">
            包含日志的一个或多个32字节非索引参数。
        </ResponseField>

        <ResponseField name="logIndex" type="string">
            logIndex属性指定块中日志项的索引或位置。它被编码为十六进制字符串。
        </ResponseField>

        <ResponseField name="topics" type="string">
            topics属性是索引日志参数的0到4个32字节主题哈希的数组。
        </ResponseField>

        <ResponseField name="transactionHash" type="string">
            transactionHash属性指定交易的哈希。
        </ResponseField>

        <ResponseField name="transactionIndex" type="string">
            transactionIndex属性指定块中事务的索引或位置。它被编码为十六进制字符串。
        </ResponseField>


    </Expandable>



  </Expandable>

</ResponseField>
